  .amount(125D)
  .currency("RUR")
  .comment("Java-SDK-Test")
  .setTestMode()
  .build();
 
// Payment Response
PaymentResponse paymentResponse = apiCommandsFacade.requestPayment(paymentRequest)

  new PaymentStatusRequest(1228221L)
